\section{Paper Overview}
%---------------------------------------------------------------------
This paper describes extensions to the SCALE code system (specifically TRITON and ORIGEN modules) for simulating fuel depletion in molten salt reactors (MSRs) and other liquid-fueled nuclear systems. Key features include modeling continuous material feeds and removals (online reprocessing), flowing fuel with delayed neutron precursor transport, and salt treatment processes. The paper validates the methodology against the Molten Salt Reactor Experiment (MSRE) and presents test cases for various MSR concepts.

\subsection{Phase 2: Reactor Model Construction}
\begin{enumerate}[label=\textbf{2.\arabic*}]
  \item \textbf{Define the reactor geometry} in SCALE input format.
    \begin{itemize}
      \item MSRE-like single-fluid reactor with graphite moderator.
      \item Salt composition: LiF-BeF$_2$-ZrF$_4$-UF$_4$ (as specified).
    \end{itemize}
  \item \textbf{Define material compositions.}
    \begin{itemize}
      \item Fuel salt isotopic composition (enrichment, carrier salt).
      \item Moderator and structural materials.
    \end{itemize}
\end{enumerate}

\subsection{Phase 3: Depletion with Material Flows}
\begin{enumerate}[label=\textbf{3.\arabic*}]
  \item \textbf{Set up TRITON depletion sequence} with the liquid-fuel extensions.
  \item \textbf{Define continuous feed/removal streams.}
    \begin{itemize}
      \item Fission gas removal (Kr, Xe) with specified removal rates.
      \item Noble metal removal.
      \item Fuel salt feed (makeup uranium or thorium).
      \item Protactinium removal (for Th-U cycle).
    \end{itemize}
  \item \textbf{Define depletion parameters.}
    \begin{itemize}
      \item Power level, irradiation time steps, predictor-corrector settings.
    \end{itemize}
  \item \textbf{Run the depletion calculation.}
\end{enumerate}

\subsection{Phase 4: Post-Processing and Analysis}
\begin{enumerate}[label=\textbf{4.\arabic*}]
  \item \textbf{Extract nuclide inventories} vs.\ time.
  \item \textbf{Compute $k_{\text{eff}}$ vs.\ burnup.}

  \item \textbf{Compute neutron spectra} at selected time points.
