\section{Paper Overview}
%---------------------------------------------------------------------
This paper analyzes the bias that arises when Markov state models (MSMs) are estimated from short, non-equilibrium molecular dynamics trajectories. The authors derive analytical expressions for the estimation bias in transition probabilities and relaxation timescales, propose correction methods (e.g., using the Koopman reweighting approach or augmented estimators), and validate their corrections on synthetic multi-state systems and on alanine dipeptide MD data using PyEMMA.

\subsection{Phase 2: MSM Estimation}
\begin{enumerate}[label=\textbf{2.\arabic*}]
  \item \textbf{Standard MSM estimation.}
    \begin{itemize}
      \item Count transitions $C_{ij}(\tau)$ from trajectory data.
      \item Estimate transition matrix $T_{ij}(\tau) = C_{ij}(\tau) / \sum_j C_{ij}(\tau)$.
      \item Compute implied timescales $t_k = -\tau / \ln \lambda_k$.
    \end{itemize}
  \item \textbf{Bias analysis.}
    \begin{itemize}
      \item Compare estimated quantities to exact (known) values for synthetic systems.
      \item Measure bias as a function of trajectory length and number of trajectories.
    \end{itemize}
\end{enumerate}

\subsection{Phase 3: Bias Correction Methods}
\begin{enumerate}[label=\textbf{3.\arabic*}]
  \item \textbf{Implement Koopman reweighting.}
    \begin{itemize}
      \item Estimate stationary distribution weights from non-equilibrium data.
      \item Use reweighted count matrix for MSM estimation.
    \end{itemize}
  \item \textbf{Implement augmented estimator} corrections as described.
  \item \textbf{Compare corrected vs.\ uncorrected estimates} across parameter regimes.
\end{enumerate}
