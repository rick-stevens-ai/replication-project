\section{Paper Overview}
%---------------------------------------------------------------------
This paper presents a supervised metagenome binning pipeline integrated into the PATRIC bioinformatics platform. The workflow takes raw metagenomic sequencing reads, assembles them into contigs using metaSPAdes, bins contigs into metagenome-assembled genomes (MAGs) using a reference-guided (supervised) approach, annotates bins using RASTtk, and evaluates genome quality using EvalG (completeness/contamination via universal marker genes) and EvalCon (annotation consistency). The pipeline is validated on synthetic communities and real environmental metagenomes.

\section{Detailed Workflow}
%---------------------------------------------------------------------

\subsection{Phase 1: Data Acquisition}
\begin{enumerate}[label=\textbf{1.\arabic*}]
  \item \textbf{Download metagenomic sequencing reads} from NCBI SRA.
    \begin{itemize}
      \item Specific SRA accession numbers referenced in the paper.
      \item Both synthetic community (mock) and real environmental samples.
    \end{itemize}
  \item \textbf{Quality control reads.}
    \begin{itemize}
      \item Adapter trimming (Trimmomatic or fastp).
      \item Quality filtering (Q $\geq$ 20).
      \item Remove host contamination if applicable.
    \end{itemize}
\end{enumerate}

\subsection{Phase 2: Assembly}
\begin{enumerate}[label=\textbf{2.\arabic*}]
  \item \textbf{Assemble reads with metaSPAdes.}
    \begin{itemize}
      \item \texttt{metaspades.py -1 reads\_1.fq -2 reads\_2.fq -o assembly\_output}
      \item Multiple $k$-mer sizes (default: 21, 33, 55, 77).
    \end{itemize}
  \item \textbf{Assess assembly quality} (QUAST or basic N50/contig count statistics).
\end{enumerate}

\subsection{Phase 3: Supervised Binning}
\begin{enumerate}[label=\textbf{3.\arabic*}]
  \item \textbf{Map reads back to contigs} (BWA-MEM or Bowtie2) to compute coverage.
  \item \textbf{Taxonomic classification of contigs.}
    \begin{itemize}
      \item BLAST or Kraken2 against a reference database.
      \item Assign contigs to taxonomic bins based on classification + coverage + composition.
    \end{itemize}
  \item \textbf{Supervised binning using SEEDtk tools.}
    \begin{itemize}
      \item Reference-guided assignment using PATRIC genome database.
      \item Iterative refinement of bins.
    \end{itemize}
\end{enumerate}

\subsection{Phase 4: Annotation and Quality Assessment}
\begin{enumerate}[label=\textbf{4.\arabic*}]
  \item \textbf{Annotate each bin with RASTtk.}
    \begin{itemize}
      \item Gene calling, functional annotation, subsystem classification.
    \end{itemize}
  \item \textbf{Evaluate genome quality.}
    \begin{itemize}
      \item \textbf{EvalG}: completeness and contamination via universal single-copy marker genes.
      \item \textbf{EvalCon}: consistency of annotation with expected genome features.
      \item Apply MIMAG quality standards (high: $> 90\%$ complete, $< 5\%$ contamination).
    \end{itemize}
\end{enumerate}

\subsection{Phase 5: Validation}
\begin{enumerate}[label=\textbf{5.\arabic*}]
  \item Compare MAG quality against known reference genomes (for synthetic communities).
  \item Compare binning performance against MetaBAT2, MaxBin2, CONCOCT.
  \item Reproduce quality summary tables from the paper.
\end{enumerate}
